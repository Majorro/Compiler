routine CastingAndRecord(
    r: record {
        var ifield : integer;
        var rfield : real;
        var bfield : boolean;
    } end
): record {
       var ifield : integer;
       var rfield : real;
       var bfield : boolean;
   } end
is
    r.ifield := 0;
    r.rfield := 0.0;
    r.bfield := false;
    return r;
end;
