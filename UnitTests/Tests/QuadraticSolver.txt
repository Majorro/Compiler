routine BabylonianSqrt(x: real): real is
    var start is 0;
    var stop is 10000;
    
    var n is 1.0;
    for i in start .. stop loop
        n := (n + (x / n)) * 0.5;
    end;
    return n;
end;

routine QuadraticSolver(a: real, b: real, c: real): integer is
    var d is BabylonianSqrt((b * b) - (4 * a * c));
    return a + 1;
end;
