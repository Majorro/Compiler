routine getSumInRange(a: integer, b: integer): integer is
    var sum is 0;
    var start is 0;
    var end is 0;
    
    if a > b then
        start := b;
        end := a;
    else
        start := a;
        end := b;
    end;
    
    for i in start .. end loop
        sum := sum + i;     
    end;
    
    return sum;
end;

routine main(): integer is
    var a is getSumInRange(1, 10);
    return a;
end;
